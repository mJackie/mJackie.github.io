PHP网页版简单计算器

Tech
  • Php

php写的一个网页版简单计算器，实现了基本的加、减、乘、除、求余运算。

<?php
    //error_reporting(E_ALL & ~[E_NOTICE]);
    if(isset($_POST['sub'])) {
        //echo "用户提交了";

        $bz = true;
        $errormess = "有以下问题：<br>";

        if($_POST[num1] == "") {
            $bz = false;
            $errormess .= "第一个数不能为空<br>";
        } else {
            if(!is_numeric($_POST[num1])) {
            $bz = false;
            $errormess .= "第一个数不是数字<br>";
            }
        }


        if($_POST[num2] == "") {
            $bz = false;
            $errormess .= "第二个数不能为空<br>";
        } else {
            if(!is_numeric($_POST[num2])) {
            $bz = false;
            $errormess .= "第二个数不是数字<br>";
            }
        }


        if($bz) {

            $sum = "";

            //判断运算符
            switch($_POST[ysf]) {
            case '+' : $sum = $_POST[num1] + $_POST[num2];break;
            case '-' : $sum = $_POST[num1] - $_POST[num2];break;
            case '*' : $sum = $_POST[num1] * $_POST[num2];break;
            case '/' : $sum = $_POST[num1] / $_POST[num2];break;
            case '%' : $sum = $_POST[num1] % $_POST[num2];break;    
            }
        }
    }
?>

<html>
    <head>
        <title>简单计算器</title>
    </head>

    <body>
        <table border="0" width="400" align="center">
        <form action="caculator.php" method="post">
            <caption><h1>简单计算器</h1></caption>
            <tr>
                <td>
                <input type="text" size="5" name="num1" value="<?php echo $_POST[num1] ?>"> <br>
                </td>

                <td>
                    <select name="ysf">
                        <option <?php if($_POST[ysf]=="+") echo "selected" ?> value="+"> + </option>
                        <option <?php if($_POST[ysf]=="-") echo "selected" ?> value="-"> - </option>
                        <option <?php if($_POST[ysf]=="*") echo "selected" ?> value="*"> * </option>
                        <option <?php if($_POST[ysf]=="/") echo "selected" ?> value="/"> / </option>
                        <option <?php if($_POST[ysf]=="%") echo "selected" ?> value="%"> % </option>
                    </select>
                </td>

                <td>
                    <input type="text" size="5" name="num2" value="<?php echo $_POST[num2] ?>"> <br>    
                </td>

                <td>
                    <input type="submit" name="sub" value="计算">
                </td>
            </tr>

            <tr>
                <td colspan="4">
                    <?php
                        if($bz) {
                            echo "计算结果：{$_POST[num1]} {$_POST[ysf]} {$_POST[num2]} =  {$sum}";
                        } else {
                        echo $errormess;
                        }

                    ?>
            </tr>
        </form>
        </table>

    </body>
</html>
